#include "leetcode.h"

class Solution {
public:
    vector<int> constructArr(vector<int>& a) {
        if(a.size() == 0) return {};

        vector<int> ans(a.size(), 0);

        ans[0] = 1;

        for(int i = 1, mult = a[0];i < a.size();i++){
            ans[i] = mult;
            mult *= a[i];
        }

        for(int i = a.size() - 2, mult = a[a.size()-1];i >= 0;i--){
            ans[i] *= mult;
            mult *= a[i];
        }

        return ans;

    }
};
