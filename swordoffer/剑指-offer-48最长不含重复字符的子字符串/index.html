#include "leetcode.h" 
//leetcode submit region begin(Prohibit modification and deletion)
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int n = s.size();

        unordered_map<char, int> mp;
        int res = 0;
        for(int i = 0, j = 0;j < s.size();j++){
            mp[s[j]]++;
            while(mp[s[j]] > 1){
                mp[s[i]]--;
                i++;
            }
            res = max(res, j - i + 1);
            //cout<<i<<" "<<j<<" "<<res<<endl;
        }
        return res;
    }
};
