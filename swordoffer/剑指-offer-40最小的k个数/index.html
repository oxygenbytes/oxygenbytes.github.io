#include "leetcode.h"

class Solution {
public:
    vector<int> getLeastNumbers(vector<int>& arr, int k) {
        int n = arr.size();
        priority_queue<int, vector<int>, less<int>> pq;

        for(int i = 0;i < n;i++){
            if(pq.size() < k){
                pq.push(arr[i]);
            }else{
                if(pq.size() && pq.top() > arr[i]) pq.pop(), pq.push(arr[i]);
            }
        }
        vector<int> ans;
        while(pq.size()) ans.push_back(pq.top()), pq.pop();

        return ans;
    }
};
