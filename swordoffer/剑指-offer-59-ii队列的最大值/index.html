class MaxQueue {
public:
    queue<int> q;
    deque<int> dq;

    MaxQueue() {

    }

    int max_value() {
        if(dq.empty()) return -1;
        else return dq.front();
    }

    void push_back(int value) {
        q.push(value);
        while(!dq.empty() && dq.back() < value){
            dq.pop_back();
        }
        dq.push_back(value);
    }

    int pop_front() {
        int t = -1;
        if(!q.empty()) t = q.front(), q.pop();
        if(!dq.empty() && dq.front() == t) dq.pop_front();
        return t;
    }
};
