class Solution {
public:
    string reverseWords(string s) {
        if(s.size() <= 0)  return s;

        int pos;
        vector<string> ans;
        for(int i = 0, j = 0;i < s.size() && j <= s.size(); j++){
            if(s[i] == ' ')   i++;
            else if(s[j] == ' ' || j == s.size()){
                ans.push_back(s.substr(i, j - i));
                i = j+1;
            }
        }
        reverse(ans.begin(), ans.end());
        string res;

        for(int i = 0;i < ans.size();i++){
            if(i == 0) res += ans[i];
            else res += ' ', res += ans[i];
        }

        return res;
    }
};
