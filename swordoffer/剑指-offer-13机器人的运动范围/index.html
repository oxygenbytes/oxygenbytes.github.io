#include "leetcode.h" 
//leetcode submit region begin(Prohibit modification and deletion)
class Solution {
public:
    int calc(int x, int y){
        int res = 0;
        while(x) res += x % 10, x /= 10;
        while(y) res += y % 10, y /= 10;
        return res;
    }
    int m, n, k;
    int movingCount(int _m, int _n, int _k) {
        result = 0;
        m = _m, n = _n, k = _k;

        vector<vector<bool>> visited(m, vector<bool>(n, false));
        dfs(0, 0, visited);
        return result;
    }
    int result;
    int dx[4] = {-1, 0, 1, 0};
    int dy[4] = {0, 1, 0, -1};
    void dfs(int x, int y, vector<vector<bool>>& visited){
        if(visited[x][y])
            return ;
        if(calc(x, y) > k){
            visited[x][y] = true;
            return ;
        }
        result ++;
        visited[x][y] = true;
        for(int i = 0;i < 4;i++){
            int a = x + dx[i], b = y + dy[i];
            if(a >= 0 && b >= 0 && a < m && b < n){
                dfs(a, b, visited);
            }
        }
    }
};
