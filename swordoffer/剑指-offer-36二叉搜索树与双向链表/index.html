class Solution {
public:
    Node *pre, *head;
    Node* treeToDoublyList(Node* root) {
        if(!root) return nullptr;
        dfs(root);
        head->left = pre;
        pre->right = head;
        return head;
    }

    void dfs(Node* root){
        if(!root) return ;
        dfs(root->left);

        if(pre) pre->right = root;
        else head = root;
        root->left = pre;
        pre = root;

        dfs(root->right);
    }
};
