#include "leetcode.h"

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *ha, ListNode *hb) {
        if(!ha || !hb){
            return nullptr;
        }
        int lena = 0, lenb = 0;
        ListNode *p = ha, *q = hb;

        while(p)  p = p->next, lena++;
        while(q)  q = q->next, lenb++;


        if(lena > lenb){
            p = ha;
            q = hb;
        }else{
            p = hb;
            q = ha;
        }

        int gap = abs(lena-lenb);
        while(gap--){
            p = p->next;
        }

        while(p && q){
            if(p == q) return p;
            p = p->next;
            q = q->next;
        }

        return nullptr;

    }
};
