#include "leetcode.h" 
//leetcode submit region begin(Prohibit modification and deletion)
class Solution {
public:
    vector<vector<int>> findContinuousSequence(int target) {
        vector<vector<int>> res;

        vector<int> temp;

        for(int i = 1, j = 2;i < j && j < target;) {
            int ans = (i + j) * (j - i + 1) / 2;

            if (ans == target) {
                temp.clear();
                for (int k = i; k <= j; k++)
                    temp.push_back(k);
                res.push_back(temp);
                i++;j++;
            } else if (ans < target) {
                j++;
            } else
                i++;
        }
        return res;
    }
};
