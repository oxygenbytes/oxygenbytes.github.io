#include "leetcode.h" 
//leetcode submit region begin(Prohibit modification and deletion)
class Solution {
public:
    int n, m;
    bool exist(vector<vector<char>>& board, string word) {

        m = board.size();
        n = board[0].size();

        if(word.empty()) return true;
        if(board.empty() || board[0].empty()) return false;

        for(int i = 0;i < m;i++){
            for(int j = 0; j < n;j++){
                if(board[i][j] == word[0]){
                    if(dfs(board,word, i, j, 0)) return true;
                }
            }
        }
        return false;
    }

    bool dfs(vector<vector<char>>& board,string& word, int x, int y, int cur){
        if(board[x][y] != word[cur]) return false; // æ¯”è¾ƒæœ€åä¸€ä¸ªå­—ç¬¦åï¼Œå†æ ¹æ®curåˆ¤æ–­
        if(cur == word.size() - 1) return true;

        int dx[4] = {-1, 0, 1, 0};
        int dy[4] = {0, 1, 0, -1};

        board[x][y] = '.'; // ä¸å›æº¯

        for(int i = 0;i < 4;i++){
            int a = x + dx[i];
            int b = y + dy[i];

            if(a >= 0 && a < m && b >= 0 && b < n){
                if(dfs(board, word, a, b, cur + 1)){
                    return true;
                }
            }
        }
        board[x][y] = word[cur];
        return false;
    }
};
