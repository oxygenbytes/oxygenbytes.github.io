class Solution2 {
public:
    int findMin(vector<int>& nums) {
        int n = nums.size();
        if(!n) return 0;

        if(nums[n-1] > nums[0]) return nums[0];

        while(--n){
            if(n >= 0 && nums[n] == nums[0]) n--;
            else break;
        }

        int l = 0, r = n;
        while(l < r){
            int mid = l + r >> 1;
            if(nums[mid] < nums[0]) r = mid; // æ³¨æ„è¿™é‡Œçš„ç‰¹å¾ï¼Œæ˜¯nums[mid] < nums[0]
            else l = mid + 1;
        }
        return nums[l];
    }
};
