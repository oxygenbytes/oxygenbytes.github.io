/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution2 {
public:
    bool hasCycle(ListNode *head) {
        unordered_set<ListNode*> us;

        while(head){
            if(us.count(head)) return true;
            us.insert(head);
            head = head->next;
        }
        return false;
    }
};

class Solution {
public:
    bool hasCycle(ListNode *head) {
        auto fast = head;
        auto slow =  head;

        while(fast){
            if(!fast->next) return false;
            fast = fast->next->next;
            slow = slow->next;
            if(fast == slow) return true;
        }
        return false;
    }
};


class Solution3 {
public:
    bool hasCycle(ListNode *head) {
        if(!head) return false;
        auto fast = head, slow = head;
        while(fast){
            if(fast->next) fast = fast->next->next, slow = slow->next;
            else return false;
            if(fast == slow) return true;
        }
        return false;
    }
};
