/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* insertionSortList(ListNode* head) {
        if(!head || !head->next) return head;

        auto dummy = new ListNode(-1); // éå†næ¬¡ï¼Œæ¯æ¬¡æ‰¾åˆ°ç›®æ ‡èŠ‚ç‚¹ï¼Œå¹¶æ›´æ–°curæ‰€åœ¨é“¾è¡¨
        auto pre = head;


        while(pre){
            auto ptr = pre->next;
            auto cur = dummy;

            while(cur->next && cur->next->val <= pre->val){
                cur = cur->next;
            }
            pre->next = cur->next;
            cur->next = pre;
            pre = ptr;
        }

        return dummy->next;
    }
};
