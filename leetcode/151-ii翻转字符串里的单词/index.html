class Solution {
public:
    string reverseWords(string s) {
        int k = 0;
        for(int i = 0;i < s.size();i++){
            while(i < s.size() && s[i] == ' ') i ++ ; // find word begin == i
            if(i == s.size()) break;
            int j = i;
            while(j < s.size() && s[j] != ' ') j ++; // find word end == j

            reverse(s.begin() + i, s.begin() + j); // reverse
            if(k) s[k++] = ' '; // skip head ' '
            while(i < j) s[k++] = s[i++]; // copy
        }

        s.erase(s.begin() + k, s.end()); // erase end of them
        reverse(s.begin(), s.end()); // reverse
        return s;
    }
};
