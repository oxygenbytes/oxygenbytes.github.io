class Solution {
public:
    int reverse(int x) {
        if(!x) return 0;
        bool flag = x > 0 ? true : false;
        x = abs(x);
        long long int res = 0;

        while(x){
            res = res * 10;
            res += x % 10;
            x /= 10;
        }
        if(res > INT_MAX || res < INT_MIN) return 0;

        return flag ? res : -res;
    }
};
