#include "leetcode.h" 
//leetcode submit region begin(Prohibit modification and deletion)
class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        // 10^6 ==> (0, 2^20)
        unordered_set<int> prime({2, 3, 5, 7, 11, 13, 17, 19});

        int res = 0;
        for(int i = L;i <= R;i++){
            int s = 0;
            for(int k = i; k;k >>= 1) s += k & 1;
            if(prime.count(s)) res++;
        }
        return res;
    }
};
