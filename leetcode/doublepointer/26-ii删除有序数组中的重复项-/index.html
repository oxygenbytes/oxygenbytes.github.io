#include "leetcode.h"

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if(nums.empty()) return 0;
        int i = 0, j = 0;
        while(j < nums.size()){
          if(i == 0 || nums[j] != nums[i-1]){
            nums[i++] = nums[j++];
          }else
            j++;
        }
        return i;
    }
};
