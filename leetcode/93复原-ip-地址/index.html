#include "leetcode.h" 
//leetcode submit region begin(Prohibit modification and deletion)
class Solution {
public:
    vector<string> ans;
    string path;
    vector<string> restoreIpAddresses(string s) {
        if(s.empty() || s.size() > 12) return {};
        dfs(s, 0, 0);
        return ans;
    }

    void dfs(string& s, int idx, int cnt){
        if(cnt == 4 && idx >= s.size()){
            ans.push_back(path);
            return ;
        }

        for(int len = 1;len <= 3 && idx + len <= s.size();len++){
            string cur = s.substr(idx, len);
            if(cur.size() > 1 && cur[0] == '0') continue; // ä¸€ä¸ª0æ˜¯å¯ä»¥çš„
            int num = stoi(cur);

            if(num >= 0 && num <= 255){
                string temp = path;
                path += s.substr(idx, len);
                if(idx + len != s.size()) path += '.';
                dfs(s, idx + len, cnt + 1);
                path = temp;
            }
        }
    }
};
