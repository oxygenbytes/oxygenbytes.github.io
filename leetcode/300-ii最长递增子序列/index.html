class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        if(n == 0) return 0;
        // ä»¥nums[i]ç»“å°¾çš„çš„æœ€é•¿ä¸Šå‡å­åºåˆ—
        vector<int> dp(n, 1);

        for(int i = 0;i < n;i++){
            for(int j = 0;j < i;j++){
                if(nums[j] < nums[i]){
                    dp[i] = max(dp[i],dp[j]+1);
                }
            }
        }

        return *max_element(dp.begin(), dp.end());
    }
};
