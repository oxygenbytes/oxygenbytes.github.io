class Solution {
public:
    bool repeatedSubstringPattern(string s) {
        int n = s.size();
        // nextæ•°ç»„
        vector<int> next(n);
        next[0] = -1;
        int j = -1;
        // æ³¨æ„iä»1å¼€å§‹
        for(int i = 1;i < n;i++){
          while(j > -1 && s[i] != s[j+1]) j = next[j];
          if(s[i] == s[j+1]) j++;
          next[i] = j;
        }
        // next[n-1] - 1 å‡1æ˜¯å› ä¸ºä»-1å¼€å§‹è®¡ç®—
        return next[n-1] != -1 && n % (n - next[n-1] - 1) == 0;

};
